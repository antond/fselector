def to_scale(min=0.0, max=1.0)
  if (min >= max)
    abort "[#{__FILE__}@#{__LINE__}]: "+
          "min must be smaller than max"
  end
  
  old_min = self.min
  old_max = self.max

  self.collect do |v|
    if old_min == old_max
      max
    else
      min + (v-old_min)*(max-min)/(old_max-old_min)
    end
  end
end

def to_zscore
  ave = self.ave
  sd = self.sd

  return self.collect { |v| (v-ave)/sd }
end

def var
  u = self.ave
  v2 = self.inject(0.0) { |v, i| v+(i-u)*(i-u) }
  
  v2/(self.size-1)
end
