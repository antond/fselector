def critchi(p, df)
  minchisq = 0.0
  maxchisq = CHI_MAX
  
  chisqval = nil
  
  if p <= 0.0
    return maxchisq
  else
    if p >= 1.0
      return 0.0
    end
  end
  
  chisqval = df / Math.sqrt(p);    # fair first value

  while (maxchisq - minchisq) > CHI_EPSILON
    if pochisq(chisqval, df) < p
      maxchisq = chisqval
    else
      minchisq = chisqval
    end
    
    chisqval = (maxchisq + minchisq) * 0.5
   end
   
   return chisqval
end

def pochisq(x, df)
  a, y, s = nil, nil, nil
  e, c, z = nil, nil, nil
  
  even = nil # True if df is an even number
  
  if x <= 0.0 or df < 1
    return 1.0
  end
  
  a = 0.5 * x
  even = ((df & 1) == 0)
  
  if df > 1
    y = ex(-a)
  end
  
  s = even ? y : (2.0 * poz(-Math.sqrt(x)))
  
  if df > 2
    x = 0.5 * (df - 1.0)
    z = even ? 1.0 : 0.5
    
    if a > BIGX
      e = even ? 0.0 : LOG_SQRT_PI
      c = Math.log(a)
      
      while z <= x
        e = Math.log(z) + e
        s += ex(c * z - a - e)
        z += 1.0
      end
      
      return s 
    else
      e = even ? 1.0 : (I_SQRT_PI / Math.sqrt(a))
      c = 0.0
      
      while (z <= x)
        e = e * (a / z)
        c = c + e
        z += 1.0
      end
      
      return c * y + s
    end
  else
    return s
  end

end
