def calculate(n11_,n12_,n21_,n22_)
  n11_ *= -1 if n11_ < 0
  n12_ *= -1 if n12_ < 0
  n21_ *= -1 if n21_ < 0 
  n22_ *= -1 if n22_ < 0 
  n1_     = n11_ + n12_
  n_1     = n11_ + n21_
  n       = n11_ + n12_ + n21_ + n22_
  prob    = exact(n11_,n1_,n_1,n)
  left    = @sless
  right   = @slarg
  twotail = @sleft + @sright
  twotail = 1 if twotail > 1
  values_hash = { :left =>left, :right =>right, :twotail =>twotail }
  return values_hash
end
